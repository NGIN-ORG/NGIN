if(TARGET NGIN::Core)
  return()
endif()

set(_ngin_core_source_dir "${CMAKE_CURRENT_LIST_DIR}/../../NGIN.Core")
set(_ngin_base_source_dir "${CMAKE_CURRENT_LIST_DIR}/../../Dependencies/NGIN/NGIN.Base")
set(_ngin_log_source_dir "${CMAKE_CURRENT_LIST_DIR}/../../Dependencies/NGIN/NGIN.Log")
if(NOT EXISTS "${_ngin_core_source_dir}/CMakeLists.txt")
  message(FATAL_ERROR "Packages/NGIN.Core requires NGIN.Core/")
endif()
if(NOT EXISTS "${_ngin_base_source_dir}/CMakeLists.txt")
  message(FATAL_ERROR "Packages/NGIN.Core requires Dependencies/NGIN/NGIN.Base")
endif()
if(NOT EXISTS "${_ngin_log_source_dir}/CMakeLists.txt")
  message(FATAL_ERROR "Packages/NGIN.Core requires Dependencies/NGIN/NGIN.Log")
endif()

set(NGIN_BASE_SOURCE_DIR "${_ngin_base_source_dir}" CACHE PATH "" FORCE)
set(NGIN_LOG_SOURCE_DIR "${_ngin_log_source_dir}" CACHE PATH "" FORCE)
set(NGIN_CORE_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(NGIN_CORE_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
add_subdirectory("${_ngin_core_source_dir}" "${CMAKE_CURRENT_BINARY_DIR}/_ngin_core" EXCLUDE_FROM_ALL)
